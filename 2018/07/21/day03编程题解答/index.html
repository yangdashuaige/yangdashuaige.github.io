package day04;
import java.util.Scanner;
public class demo01 {
public static void main(String[] args) {
System.out.println(“输入球下落的高度”);
Scanner scanner=new Scanner(System.in);
double height=scanner.nextDouble();
System.out.println(“输入第几次落地的值n”);
int n=scanner.nextInt();
double sum=0; //经过的总路程
double tan=0; //第n次弹起的高度
for(int i=0;i<n;i++)
{
tan=height/2;
sum+=height+tan;
height=tan;
}
System.out.println(“第”+n+”次落地共经过”+sum+”米；弹起的高度为：”+tan+”米。”);

}

}

package day04;

import java.util.Scanner;

public class demo12 {

public static void main(String[] args) {
    Scanner scanner=new Scanner(System.in);
    System.out.println("请输入用户名：");
    String name=scanner.next();//方法返回的是字符串
    System.out.println("请输入密码:");
    String pwd=scanner.next();
    //验证码  随机6位数
    int num=(int)(Math.random()*1000000);
    System.out.println("验证码为："+num+"，请输入：");
    int yan=Integer.parseInt(scanner.next());//有问题。字符串-》基本数据类型
    if(num==yan)
    {
        //判断用户名和密码   基本数据类型的变量名就是变量的值，其他类型的不是
        if(name.equals("杨环"))  //这边使用equals方法来判断
        {
            //登录成功
            boolean flag=false;
            for(int i=1;i<=5;i++)
            {
                if(pwd.equals("123"))
                {
                    flag=true;
                    break;
                }
                else {
                    System.out.println("密码错误，还有"+(5-i)+"次机会。");
                    pwd=scanner.next();
                }
            }
            if(flag){
                System.out.println("登录成功");    
            }
            else {
                System.out.println("五次密码不正确，系统登录失败");
            }                
        }
        else{
            //登录失败
            System.out.println("用户名错误");
        }
    }
    else{
        System.out.println("验证码错误");
    }
}

}
