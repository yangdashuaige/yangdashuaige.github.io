  1. 具体的实现方法

    @Service
public class GoodsServiceImpl implements GoodsService {
@Autowired
private GoodsMapper mapper; //商品的mapper

@Autowired
private GoodsCatoryMapper catoryMappermapper;//商品类别的mapper

@Override
public PageInfo<Goods> getPageInfor(int curPage,SearchBean searchBean) {
//         查询的起始页
//          每页的条数  5条
//            总条数
      int count =mapper.selectcount(searchBean);
//          拦截添加分页
    PageHelper.startPage(curPage,5);
    List<Goods> goodsList=mapper.selectAllGoods(searchBean);//按条件查询全部的商品
    PageInfo<Goods> pageInfo=new PageInfo<>(goodsList);
    //当前页
    pageInfo.setPageNum(curPage);
    //每页显示的条数
    pageInfo.setPageSize(5);
    //总条数
    pageInfo.setTotal(count);
    return pageInfo;
}

@Override
public List<GoodsCatory> getAllGoodsCatory() {
    return catoryMappermapper.selectAllCatory();
}

@Override
public Goods getGoodsById(int id) {
    return mapper.selectById(id);
}

@Override
public boolean removeGooods(int id) {
    return mapper.deleteById(id);
}

//数据的数据不存在 返回false
//有一个删除不成功,返回false
//    多删除,多delete
@Transactional(isolation = Isolation.DEFAULT,propagation = Propagation.REQUIRED)
@Override
public boolean removeManyGooods(int[] gou) {
    //
     if(gou.length<=0){
         return  false;
     }
     boolean b=true;//默认删除成功
    for (int i=0;i<gou.length;i++){
        b&=mapper.deleteById(gou[i]);
    }
    return b;
    }

}

  3. Control层

import javax.servlet.http.HttpServletRequest;
import java.util.Arrays;
import java.util.List;
@RequestMapping("/goods") //访问路径
@Controller//实例化处理器
public class GoodsControl {

@Autowired
 private GoodsService goodsService;

//method = RequestMethod.GET 限定请求的方式
@RequestMapping(value = "/list.action")
//    RequestParam 如果为空,添加默认值
//出现400的原因,参数没有办法封装实体类,类型转换不正确    String
public String main(@RequestParam(value ="currentPage",defaultValue = "1") int currentPage,SearchBean bean, Model model, HttpServletRequest request){//请求的参数名和属性一致
     List<GoodsCatory> goodsCatoryList=goodsService.getAllGoodsCatory();
      PageInfo<Goods> pageInfo= goodsService.getPageInfor(currentPage,bean);
      model.addAttribute("catoryList",goodsCatoryList);
      model.addAttribute("pageBean",pageInfo);
//      <input name="maxprice" type="text" class="allInput"  value="${searchBean.maxprice}"/>
//     只要属性值一致,那么就相当于放在model里面 (参数 封装请求的参数,jsp的域对象也会在此取)
//    表单回显 (pojo属性名     表的类名一致  mybatis      form 表单的 name一致   spring会封装       value值也要一致  )
       return "/jsp/list.jsp";
}

    @RequestMapping("/goodInfor.action")
  public String GoodInfor(int id,Model model){
     Goods goods=goodsService.getGoodsById(id);//关于查询+数据库磁盘(缓存(redis ))  [体现]
     model.addAttribute("goods",goods);
     return "/jsp/GoodInfor.jsp";
 }


   @RequestMapping("/delete.action") //变量,对象封装
   public String delete(int[] gou){
      boolean b=goodsService.removeManyGooods(gou);
      if (b){
          return "redirect:/goods/list.action";
      }else{
          return "redirect:/goods/list.action?error=1";//删除不成功
      }
   }


}
